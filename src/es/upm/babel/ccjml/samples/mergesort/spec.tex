\begin{ctadsol}
\label{especificacion-inicial}
\nombrectad{MergeSortComp}

\operaciones
\operacion{SetLeft}{ANY[i]}
\operacion{SetRight}{ANY[i]} 
\operacion{GetResult}{ANY[o]}

\semdom
  \estado{MergeSortComp =  (right : ANY \por left : ANY) }
  \tipo{EOD $\in$ ANY}

  \invariante{EOD \not= nil \land (\forall o \in ANT \st EOD > o)}
  \inicial{self}{self.left = nil \land self.right = nil}

  \pre{(obj \not= nil)}
  \cpre{(self.left = nil)}
  \nomop{SetLeft(obj)}
  \post{self\sal.left = obj \land self\sal.right = self\ent.right}

  \pre{(obj \not= nil)}
  \cpre{(self.right = nil)}
  \nomop{SetRight(obj)}
  \post{self\sal.right = obj \land self\sal.left = self\ent.left}

  \pre{\true}
  \cpre{self.right \not= nil \land self.left \not= nil}
  \nomop{getResult(res)}
  \post{(res = self.left \land self.left <= self.right) 
        \lor
        (res = self.right \land self.right < self.left)
        \land self\sal.left = self\ent.left
        \land self\sal.right = self\ent.right}

\end{ctadsol}
